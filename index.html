class D3TwinsViewer extends HTMLElement { constructor() { super(); // Shadow DOM cerrado para encapsular el iframe const shadow = this.attachShadow({ mode: "closed" }); // Crear iframe donde se cargará el visor real this.iframe = document.createElement("iframe"); this.iframe.setAttribute("frameborder", "0"); this.iframe.style = "width:100%;height:100%;border:0;"; shadow.appendChild(this.iframe); } connectedCallback() { const viewerSrc = this.getAttribute("src"); // URL que apunta a viewer.3dtwins.tech if (!viewerSrc) { console.error("No se proporcionó src al 3d-twins-viewer"); return; } // Sustituir solo el dominio, dejando todo lo demás intacto const realLoaderURL = viewerSrc.replace(/^https?:\/\/viewer\.3dtwins\.tech/, "https://load.3dtwins.tech"); // Asignar la URL real al iframe this.iframe.src = realLoaderURL; } } // Registrar el WebComponent customElements.define("d3-twins-viewer", D3TwinsViewer);
